#include <math.h>

const int PIN_LED = 9;
float angulo = 0.0;

void setup() {
  pinMode(PIN_LED, OUTPUT);
}

void loop() {
  // Convertimos seno (-1 a 1) → (0 a 1)
  float seno = (sin(angulo) + 1.0) / 2.0;
  
  // Escalamos a 0–255
  int brillo = (int)(seno * 255);
  
  analogWrite(PIN_LED, brillo);
  
  angulo += 0.05; // Velocidad
  
  if (angulo > 2 * PI) angulo = 0;
  
  delay(15);
}
